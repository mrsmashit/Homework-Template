\begin{problem}
Evaluate the integral $\int_0^1 x^2 \, dx$.
\end{problem}

\begin{solution}
We compute
\[
\int_0^1 x^2 \, dx = \left[ \frac{x^3}{3} \right]_0^1 = \frac{1}{3}.
\]
\end{solution}
